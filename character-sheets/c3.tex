  \node (#2) [#3] #4 { \Large #5 } ;

        {\textsf{\footnotesize #6}};
}

\newdimen\mytemp
\environbodyname\mysectionbody
\NewEnviron{mysection}[3][]{
  \setlength\mytemp{\rightwidth-6mm}
%  \node [#1] (#2) #3 {X{\begin{minipage}[t]{\mytemp}\mysectionbody\end{minipage}}Z};
  \node [#1] (#2) #3 {\begin{minipage}{3in}\lipsum[3]\end{minipage}};
  \dnddecorate{#2};
}

\newcommand\newsectionenv[2][78mm]{%
  \expandafter\environbodyname\expandafter{\csname#1body\endcsname}

    % [...] name at
    \setlength\mytemp{#1}
    \node [#2,##1] (##2) ##3 {X{\begin{minipage}[t]{\mytemp}\csname#1body\endcsname\end{minipage}}Z};
    \dnddecorate{##2};
  }
}
